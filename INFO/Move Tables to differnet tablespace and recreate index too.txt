FROM User_Tables
WHERE TABLE_NAME IN ( 'REPORTS', 'REPORT_GROUPS','REPORT_TYPES','REPORT_SUB_TYPES')
UNION ALL
SELECT INDEX_NAME,TABLESPACE_NAME,STATUS
FROM User_Indexes
WHERE TABLE_NAME IN ( 'REPORTS', 'REPORT_GROUPS','REPORT_TYPES','REPORT_SUB_TYPES');

ALTER TABLE REPORTS MOVE TABLESPACE KRBCS;
